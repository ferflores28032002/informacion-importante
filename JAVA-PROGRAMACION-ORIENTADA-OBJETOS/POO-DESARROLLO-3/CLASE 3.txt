/*
    calcular el area de triangulo, rectangulo y cuadrado
 */

import java.util.*;

class Triangulo {

    double base, altura;
    Scanner entrada = new Scanner(System.in);

    public Triangulo() {
        base = 0.0;
        altura = 0.0;
    }

    void leer() {
        System.out.println("Base: ");
        base = entrada.nextDouble();
        System.out.println("Altura: ");
        altura = entrada.nextDouble();
    }

    double calcularArea() {
        double resp;
        resp = (base * altura) / 2;
        return resp;
    }

    void imprimir() {
        System.out.println("El area del Triangulo es: " + calcularArea());
    }
}

//CLASE CUADRADO
class Cuadrado {

    double lado;
    Scanner entrada = new Scanner(System.in);

    public Cuadrado() {
        lado = 0.0;
    }

    void lectura() {
        System.out.println("Lado: ");
        lado = entrada.nextDouble();
    }

    double calcularArea() {
        return lado * lado;
    }
}

//CLASE RECTANGULO
class Rectangulo {

    double base, altura;
    Scanner entrada = new Scanner(System.in);

    public Rectangulo() {
        base = 0.0;
        altura = 0.0;
    }

    void leer() {
        System.out.println("Base: ");
        base = entrada.nextDouble();
        System.out.println("Altura: ");
        altura = entrada.nextDouble();
    }

    void calcularArea() {
        System.out.println("El area del triangulo es: " + (base * altura));
    }

}

public class Figuras {
